## Database Query Patterns

```sql
-- Worker Dashboard Queries

-- 1. Fetch Worker Profile
SELECT * FROM worker_profiles 
WHERE user_id = $user_id;

-- 2. Fetch Applications
SELECT a.*, p.* 
FROM applications a
JOIN projects p ON a.project_id = p.id
WHERE a.worker_id = $user_id
ORDER BY a.created_at DESC
LIMIT 10;

-- 3. Fetch Contracts
SELECT * FROM contracts 
WHERE worker_id = $user_id 
AND status IN ('active', 'pending');

-- 4. Fetch Earnings
SELECT SUM(net_amount) as total
FROM payments
WHERE payment_to = $user_id 
AND status = 'completed';

-- 5. Fetch Available Projects
SELECT * FROM projects
WHERE status = 'open'
ORDER BY created_at DESC
LIMIT 20;


-- Client Dashboard Queries

-- 1. Fetch Projects
SELECT * FROM projects
WHERE client_id = $user_id
ORDER BY created_at DESC;

-- 2. Count Applications per Project
SELECT COUNT(*) FROM applications
WHERE project_id = $project_id;

-- 3. Fetch Applications
SELECT a.*, u.*
FROM applications a
JOIN users u ON a.worker_id = u.id
WHERE a.project_id IN ($project_ids)
ORDER BY a.created_at DESC;

-- 4. Fetch Contracts
SELECT * FROM contracts
WHERE project_id IN ($project_ids);

-- 5. Calculate Spending
SELECT SUM(contract_amount) as total
FROM contracts
WHERE project_id IN ($project_ids);
```
